        // Función de copiado optimizada para Iframe y dispositivos móviles
        function copyData(textToCopy, buttonId) {
            const tempInput = document.createElement("input");
            tempInput.value = textToCopy;
            document.body.appendChild(tempInput);
            
            tempInput.select();
            tempInput.setSelectionRange(0, 99999);
            
            try {
                document.execCommand("copy");
                showNotificationToast();
                triggerButtonSuccessState(buttonId);
            } catch (err) {
                console.error("Error al copiar texto: ", err);
            }
            
            document.body.removeChild(tempInput);
        }

        // Mostrar Toast Flotante
        function showNotificationToast() {
            const toast = document.getElementById("toast");
            toast.classList.remove("opacity-0", "pointer-events-none", "scale-90");
            toast.classList.add("opacity-100", "scale-100");
            
            setTimeout(() => {
                toast.classList.remove("opacity-100", "scale-100");
                toast.classList.add("opacity-0", "pointer-events-none", "scale-90");
            }, 3500);
        }

        // Animación temporal en el botón presionado
        function triggerButtonSuccessState(buttonId) {
            const btn = document.getElementById(buttonId);
            const originalHTML = btn.innerHTML;
            
            btn.innerHTML = '<i class="fas fa-check mr-1"></i> ¡Listo!';
            btn.classList.remove("bg-gold-500", "hover:bg-gold-600", "text-navy-950");
            btn.classList.add("bg-emerald-600", "hover:bg-emerald-700", "text-white");
            
            setTimeout(() => {
                btn.innerHTML = originalHTML;
                btn.classList.remove("bg-emerald-600", "hover:bg-emerald-700", "text-white");
                btn.classList.add("bg-gold-500", "hover:bg-gold-600", "text-navy-950");
            }, 2000);
        }
    </script>
